00293

00296 
00297     s[0] = &r1[0];

00299     s[2] = &r3[0];
00300     s[3] = &r4[0];
00301 
00302     register int32_t i,j,p,jj;
00303     for(i=0;i<4;i++) {
00304         for(j=0;j<4;j++) {
00305             s[i][j] = m.element(i,j);
00306             if(i==j) s[i][j+4] = 1.0;
00307             else     s[i][j+4] = 0.0;
00308         }
00309     }
00310     T scp[4];
00311     for(i=0;i<4;i++) {
00312         scp[i] = T(fabs(s[i][0]));
00313         for(j=1;j<4;j++)
00314             if(T(fabs(s[i][j])) > scp[i]) scp[i] = T(fabs(s[i][j]));
00315         if(scp[i] == 0.0) return minv; // singular matrix!
00316     }
00317 
00318     int32_t pivot_to;
00319     T scp_max;
00320     for(i=0;i<4;i++) {
00321         // select pivot row
00322         pivot_to = i;
00323         scp_max = T(fabs(s[i][i]/scp[i]));
00324         // find out which row should be on top
00325         for(p=i+1;p<4;p++)
00326             if (T(fabs(s[p][i]/scp[p])) > scp_max) {
00327                 scp_max = T(fabs(s[p][i]/scp[p]));
00328                 pivot_to = p;
00329             }
00330         // Pivot if necessary
00331         if(pivot_to != i) {
00332             tmprow = s[i];
00333             s[i] = s[pivot_to];
00334             s[pivot_to] = tmprow;
00335             T tmpscp;
00336             tmpscp = scp[i];
00337             scp[i] = scp[pivot_to];
00338             scp[pivot_to] = tmpscp;
00339         }
00340 
00341         T mji;
00342         // perform gaussian elimination
00343         for(j=i+1;j<4;j++) {
00344             mji = s[j][i]/s[i][i];
00345             s[j][i] = 0.0;
00346             for(jj=i+1;jj<8;jj++)
00347                 s[j][jj] -= mji*s[i][jj];
00348         }
00349     }

00351 
00352     //

00354     //
00355     //  x x x x | y y y y
00356     //  0 x x x | y y y y
00357     //  0 0 x x | y y y y
00358     //  0 0 0 x | y y y y
00359     //
00360     //  we'll back substitute to get the inverse
00361     //
00362     //  1 0 0 0 | z z z z
00363     //  0 1 0 0 | z z z z
00364     //  0 0 1 0 | z z z z
00365     //  0 0 0 1 | z z z z
