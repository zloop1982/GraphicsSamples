00090     void getResults() {
00091         // Make a pass over all timers - if any are pending results ("in flight"), then
00092         // grab the the time diff and add to the accumulator.  Then, mark the timer as 
00093         // not in flight and record it as a possible next timer to use (since it is
00094         // now unused)
00095         int32_t freeQuery = -1;
00096         for (unsigned int i = 0; i < TIMER_COUNT; i++) {
00097             // Are we awaiting a result?
00098             if (m_timeStampQueryInFlight[i]) {
00099                 GLuint available = false;
00100                 m_glGetQueryObjectuiv(m_queries[i][TIMESTAMP_QUERY_END], NV_QUERY_RESULT_AVAILABLE, &available);
00101 
00102                 // Is the result ready?
00103                 if (available) {
00104                     uint64_t timeStart, timeEnd;
00105                     m_glGetQueryObjectui64v(m_queries[i][TIMESTAMP_QUERY_BEGIN], NV_QUERY_RESULT, &timeStart);
00106                     m_glGetQueryObjectui64v(m_queries[i][TIMESTAMP_QUERY_END], NV_QUERY_RESULT, &timeEnd);
00107                 
00108                     // add the time to the accumulator
00109                     m_elapsedCycles += float(double(timeEnd - timeStart) * 1.e-6);
00110                     freeQuery = i;
00111                     m_timeStampQueryInFlight[i] = false;
00112 
00113                     // Increment the counter of start/stop cycles
00114                     m_startStopCycles++;
00115                 }
00116             }
00117         }
00118         // Use the newly-freed counters if one exists
00119         // otherwise, swap to the "next" counter

00123         }
